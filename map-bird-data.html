Prepare libraries and read in data

# libraries
library(tidyverse)
library(janitor)
library(here)
library(leaflet)
library(sf)

# read data
birds <- read_csv(here('ebird_US-CA_life_list.csv')) %>% 
  clean_names() %>% #clean col names
  filter(countable != 0) #remove obs that don't count

  leaflet() %>% 
  addTiles() %>%
